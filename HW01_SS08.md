1. Phân tích các trường hợp biên (Edge Cases)
Khi đọc một chuỗi JSON thô từ hệ thống Core Banking cũ, chúng ta cần lường trước và xử lý các trường hợp biên sau để tránh làm gián đoạn (crash) toàn bộ tiến trình:

Lỗi cấu trúc dữ liệu (Type Mismatch): Cực kỳ nguy hiểm. Ví dụ trường amount mong đợi là số nguyên/thực nhưng nhận được chuỗi "abc". Nếu dùng Jackson map trực tiếp toàn bộ chuỗi JSON thành List<TransactionDTO>, nó sẽ quăng JsonMappingException và làm sập toàn bộ tiến trình đọc. Giải pháp là duyệt qua từng JsonNode trước khi chuyển đổi.

Trường id không hợp lệ: Missing (không có key id), giá trị null, chuỗi rỗng "" hoặc chỉ chứa khoảng trắng "   ".

Trường amount vi phạm business logic: Giá trị âm (< 0), bằng không (== 0), hoặc null.

Trường status ngoài danh mục (Enum mismatch): Khác "SUCCESS" hoặc "FAILED" (ví dụ: "PENDING", "Success" - sai case chữ, hoặc null).

Trường transactionDate sai định dạng: Không theo chuẩn ISO-8601 (hoặc chuẩn mà hệ thống quy định), ví dụ 2023-13-45 hoặc null, gây ra DateTimeParseException.

import com.fasterxml.jackson.databind.JsonNode;
import com.fasterxml.jackson.databind.ObjectMapper;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.stereotype.Component;

import java.math.BigDecimal;
import java.time.LocalDateTime;
import java.time.format.DateTimeFormatter;
import java.time.format.DateTimeParseException;
import java.util.ArrayList;
import java.util.List;

// 1. TransactionDTO Record (Java 17)
public record TransactionDTO(
        String id,
        BigDecimal amount,
        String status,
        LocalDateTime transactionDate
) {}

// 2. TransactionParser Class
@Component
public class TransactionParser {

    private static final Logger log = LoggerFactory.getLogger(TransactionParser.class);
    private final ObjectMapper objectMapper;
    private static final DateTimeFormatter DATE_FORMATTER = DateTimeFormatter.ISO_DATE_TIME;

    public TransactionParser(ObjectMapper objectMapper) {
        this.objectMapper = objectMapper;
    }

    public List<TransactionDTO> parseTransactions(String rawJson) {
        List<TransactionDTO> validTransactions = new ArrayList<>();

        if (rawJson == null || rawJson.isBlank()) {
            log.warn("Raw JSON string is null or empty.");
            return validTransactions;
        }

        try {
            JsonNode rootArray = objectMapper.readTree(rawJson);

            if (!rootArray.isArray()) {
                log.error("Expected a JSON array, but got a different structure.");
                return validTransactions;
            }

            for (JsonNode node : rootArray) {
                try {
                    // Extract and validate ID
                    JsonNode idNode = node.get("id");
                    if (idNode == null || idNode.asText().isBlank()) {
                        throw new IllegalArgumentException("Missing or blank ID");
                    }
                    String id = idNode.asText();

                    // Extract and validate Amount
                    JsonNode amountNode = node.get("amount");
                    if (amountNode == null || !amountNode.isNumber() && !amountNode.isTextual()) {
                        throw new IllegalArgumentException("Invalid or missing amount format");
                    }
                    BigDecimal amount;
                    try {
                        amount = new BigDecimal(amountNode.asText());
                    } catch (NumberFormatException e) {
                        throw new IllegalArgumentException("Amount is not a valid number: " + amountNode.asText());
                    }
                    if (amount.compareTo(BigDecimal.ZERO) <= 0) {
                        throw new IllegalArgumentException("Amount must be greater than 0. Got: " + amount);
                    }

                    // Extract and validate Status
                    JsonNode statusNode = node.get("status");
                    if (statusNode == null || statusNode.asText().isBlank()) {
                        throw new IllegalArgumentException("Missing status");
                    }
                    String status = statusNode.asText();
                    if (!"SUCCESS".equals(status) && !"FAILED".equals(status)) {
                        throw new IllegalArgumentException("Invalid status: " + status + ". Must be SUCCESS or FAILED");
                    }

                    // Extract and validate Transaction Date
                    JsonNode dateNode = node.get("transactionDate");
                    if (dateNode == null || dateNode.asText().isBlank()) {
                        throw new IllegalArgumentException("Missing transaction date");
                    }
                    LocalDateTime transactionDate;
                    try {
                        transactionDate = LocalDateTime.parse(dateNode.asText(), DATE_FORMATTER);
                    } catch (DateTimeParseException e) {
                        throw new IllegalArgumentException("Invalid date format: " + dateNode.asText());
                    }

                    // If all validations pass, add to the valid list
                    TransactionDTO dto = new TransactionDTO(id, amount, status, transactionDate);
                    validTransactions.add(dto);

                } catch (Exception e) {
                    // Log the error and the problematic node, then skip to the next
                    log.error("Failed to parse transaction item. Error: {}. Item JSON: {}", 
                              e.getMessage(), node.toString());
                }
            }

        } catch (Exception e) {
            log.error("Failed to parse the root JSON tree. System will return an empty list.", e);
        }

        log.info("Successfully parsed {} valid transactions.", validTransactions.size());
        return validTransactions;
    }
}
